#include <ESP8266WiFi.h>
#include <ESP8266HTTPClient.h>
#include <WiFiClient.h>
#include <DHT.h> // Make sure to install "DHT sensor library" by Adafruit in Arduino IDE

// --- WiFi Settings ---
const char* ssid = "YOUR_WIFI_NAME";
const char* password = "YOUR_WIFI_PASSWORD";

// --- Server Settings ---
// Change this to your Render URL after deployment!
// Example: "https://project-iot-abc.onrender.com/api/sensor-data"
// If testing locally, use your laptop's IP address:
const char* serverName = "http://10.59.162.57:3000/api/sensor-data";

// --- Sensor Pins ---
#define DHTPIN D2     // Digital pin connected to the DHT sensor
#define DHTTYPE DHT11 // DHT 11 (Change to DHT22 if using that one)
#define SOIL_PIN A0   // Analog pin for soil moisture
#define PIR_PIN D3    // Digital pin for PIR motion sensor (Intrusion)

DHT dht(DHTPIN, DHTTYPE);

// Timer variables
unsigned long lastTime = 0;
// Set timer to 8 seconds (8000ms) to match the dashboard's auto-refresh rate
unsigned long timerDelay = 8000;

void setup() {
  Serial.begin(115200);
  
  // Initialize sensors
  dht.begin();
  pinMode(PIR_PIN, INPUT);

  // Connect to WiFi
  WiFi.begin(ssid, password);
  Serial.println("Connecting to WiFi");
  while(WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("");
  Serial.print("Connected to WiFi network with IP Address: ");
  Serial.println(WiFi.localIP());
}

void loop() {
  // Send an HTTP POST request every 8 seconds
  if ((millis() - lastTime) > timerDelay) {
    // Check WiFi connection status
    if(WiFi.status() == WL_CONNECTED) {
      
      // 1. Read Sensors
      float temperature = dht.readTemperature();
      float humidity = dht.readHumidity();
      
      // Read Soil Moisture (Analog 0-1023)
      // Map it to a percentage (0-100%). Adjust the 1023 value based on your sensor's dry/wet calibration.
      int soilAnalog = analogRead(SOIL_PIN);
      float soilMoisture = map(soilAnalog, 1023, 0, 0, 100); 
      
      // Read PIR Sensor (Intrusion)
      bool intrusion = digitalRead(PIR_PIN) == HIGH;

      // Check if any reads failed and exit early (to try again)
      if (isnan(temperature) || isnan(humidity)) {
        Serial.println("Failed to read from DHT sensor!");
        return;
      }

      // 2. Prepare HTTP connection
      WiFiClient client;
      HTTPClient http;
      http.begin(client, serverName);
      http.addHeader("Content-Type", "application/json");

      // 3. Create JSON payload
      // Format: {"temperature":28.5, "humidity":60.2, "soilMoisture":45.0, "intrusion":false}
      String jsonPayload = "{";
      jsonPayload += "\"temperature\":" + String(temperature) + ",";
      jsonPayload += "\"humidity\":" + String(humidity) + ",";
      jsonPayload += "\"soilMoisture\":" + String(soilMoisture) + ",";
      jsonPayload += "\"intrusion\":" + (intrusion ? String("true") : String("false"));
      jsonPayload += "}";

      // 4. Send the POST Request
      Serial.print("Sending Data: ");
      Serial.println(jsonPayload);
      
      int httpResponseCode = http.POST(jsonPayload);

      // 5. Print Response
      Serial.print("HTTP Response code: ");
      Serial.println(httpResponseCode);
      
      // Free resources
      http.end();
    }
    else {
      Serial.println("WiFi Disconnected");
    }
    // Reset timer
    lastTime = millis();
  }
}
